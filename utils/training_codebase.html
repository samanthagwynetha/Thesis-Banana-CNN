IMPORTS # ============================================================================ import shutil, time, json, glob from datetime import datetime from pathlib import Path import matplotlib.pyplot as plt import seaborn as sns import pandas as pd from PIL import Image, ImageDraw, ImageFont from tensorflow import keras from tensorflow.keras import layers, models, mixed_precision from tensorflow.keras.preprocessing.image import ImageDataGenerator from tensorflow.keras.preprocessing import image as keras_image from tensorflow.keras.applications import ResNet50 from tensorflow.keras.regularizers import l2 from tensorflow.keras.callbacks import ( ReduceLROnPlateau, ModelCheckpoint, LearningRateScheduler ) from sklearn.metrics import ( classification_report, confusion_matrix, precision_recall_fscore_support ) from sklearn.utils.class_weight import compute_class_weight print(f"TensorFlow: {tf.__version__}") print(f"GPU: {tf.config.list_physical_devices('GPU')}") print("✓ All imports done") # ============================================================================ # CELL 4 - 2.1 BUSINESS UNDERSTANDING # ============================================================================ print("="*70) print("2.1 BUSINESS UNDERSTANDING") print("="*70) print(""" Problem: Foliar diseases (Sigatoka, Cordana) affecting banana production - Yield losses up to 80% (Sigatoka) - Manual inspection: labor-intensive, unreliable Existing Solution (Baseline): - ResNet50 by Jiménez et al. (2025) - Accuracy: 88.90% - Training Time: ~1.5 hours (90 minutes) - Dataset: 900 images (300 per class) Research Objectives: 1. Maintain or exceed 88.90% accuracy 2. Reduce training time significantly 3. Improve computational efficiency for field deployment 4. Enable real-time disease detection Enhancements Applied: - Partial layer freezing (80 layers) - BatchNormalization + GlobalAveragePooling - AdamW optimizer + Cosine LR scheduler - FP16 mixed precision training - EarlyStopping (patience=30) - Class weights (fixes Cordana bias) """) # ============================================================================ # CELL 5 - 2.2 DATA UNDERSTANDING # ============================================================================ print("="*70) print("2.2 DATA UNDERSTANDING") print("="*70) REPO_PATH = '/content/deep-learning-banana-diseases' ORIGINAL_DIR = f"{REPO_PATH}/Data-Tesis" AUGMENTED_DIR = f"{REPO_PATH}/Imagenes-aumentadas" def count_images(directory): counts = {} for cls in os.listdir(directory): cls_path = os.path.join(directory, cls) if os.path.isdir(cls_path): counts[cls] = len([ f for f in os.listdir(cls_path) if f.lower().endswith(('.png', '.jpg', '.jpeg')) ]) return counts original_counts = count_images(ORIGINAL_DIR) augmented_counts = count_images(AUGMENTED_DIR) print("\nORIGINAL Dataset (used for training):") total_original = 0 for cls, count in sorted(original_counts.items()): print(f" {cls}: {count} images") total_original += count print(f" TOTAL: {total_original} images") print("\nAUGMENTED Dataset (available, NOT used):") total_augmented = 0 for cls, count in sorted(augmented_counts.items()): print(f" {cls}: {count} images") total_augmented += count print(f" TOTAL: {total_augmented} images") print(f"\nGRAND TOTAL AVAILABLE: {total_original + total_augmented}") print("✓ Using same 900 original images for fair comparison") # ============================================================================ # CELL 6 - 2.2.2 EXPLORATORY DATA ANALYSIS # ============================================================================ def show_sample_images(): fig, axes = plt.subplots(1, 3, figsize=(15, 5)) class_mapping = { 'Cordana': 'Cordana', 'Sanas': 'Healthy', 'SigatokaNegra': 'Black Sigatoka' } for idx, (cls, label) in enumerate(class_mapping.items()): cls_path = os.path.join(ORIGINAL_DIR, cls) images = [f for f in os.listdir(cls_path) if f.endswith(('.jpg', '.png'))] if images: img = Image.open(os.path.join(cls_path, images[0])) axes[idx].imshow(img) axes[idx].set_title(label, fontsize=12, fontweight='bold') axes[idx].axis('off') plt.suptitle('Figure 2: Sample of Banana Leaf Dataset', fontsize=14, fontweight='bold') plt.tight_layout() plt.show() show_sample_images() print("Visual Characteristics:") print(" Healthy: Uniform green, smooth, no lesions") print(" Black Sigatoka: Brown-black streaks, necrotic lesions, yellow halos") print(" Cordana: Irregular brown streaks, circular spots") # ============================================================================ # CELL 7 - 2.3 DATA PREPARATION (80/10/10 split) # ============================================================================ print("="*70) print("2.3 DATA PREPARATION") print("="*70) OUT_DIR = "/content/dataset" TRAIN_DIR = os.path.join(OUT_DIR, "train") VAL_DIR = os.path.join(OUT_DIR, "validation") TEST_DIR = os.path.join(OUT_DIR, "test") if os.path.exists(OUT_DIR): shutil.rmtree(OUT_DIR) for d in [TRAIN_DIR, VAL_DIR, TEST_DIR]: os.makedirs(d) classes = [d for d in os.listdir(ORIGINAL_DIR) if os.path.isdir(os.path.join(ORIGINAL_DIR, d))] print(f"Classes: {sorted(classes)}") rng = random.Random(SEED) total_train = total_val = total_test = 0 for cls in sorted(classes): orig_dir = os.path.join(ORIGINAL_DIR, cls) all_imgs = [(os.path.join(orig_dir, f), f) for f in os.listdir(orig_dir) if f.lower().endswith(('.png', '.jpg', '.jpeg'))] rng.shuffle(all_imgs) n = len(all_imgs) n_train = int(n * 0.8) n_val = int(n * 0.1) splits = { TRAIN_DIR: all_imgs[:n_train], VAL_DIR: all_imgs[n_train:n_train+n_val], TEST_DIR: all_imgs[n_train+n_val:] } for base, imgs in splits.items(): os.makedirs(os.path.join(base, cls), exist_ok=True) for src, name in imgs: shutil.copy2(src, os.path.join(base, cls, name)) total_train += len(splits[TRAIN_DIR]) total_val += len(splits[VAL_DIR]) total_test += len(splits[TEST_DIR]) print(f" {cls}: train={len(splits[TRAIN_DIR])}, " f"val={len(splits[VAL_DIR])}, test={len(splits[TEST_DIR])}") assert total_train == 720 and total_val == 90 and total_test == 90 print(f"\nTrain={total_train} | Val={total_val} | Test={total_test}") print("✓ 80/10/10 split verified") # ============================================================================ # CELL 8 - DATA GENERATORS # ============================================================================ IMG_SIZE = 224 BATCH_SIZE = 32 EPOCHS = 100 train_datagen = ImageDataGenerator( rescale=1./255, rotation_range=20, width_shift_range=0.2, height_shift_range=0.2, horizontal_flip=True, zoom_range=0.2, fill_mode='nearest' ) val_datagen = ImageDataGenerator(rescale=1./255) test_datagen = ImageDataGenerator(rescale=1./255) def make_gen(datagen, directory, shuffle=True): return datagen.flow_from_directory( directory, target_size=(IMG_SIZE, IMG_SIZE), batch_size=BATCH_SIZE, class_mode='categorical', shuffle=shuffle, seed=SEED if shuffle else None ) train_gen_baseline = make_gen(train_datagen, TRAIN_DIR) val_gen_baseline = make_gen(val_datagen, VAL_DIR, shuffle=False) test_gen_baseline = make_gen(test_datagen, TEST_DIR, shuffle=False) train_gen_enhanced = make_gen(train_datagen, TRAIN_DIR) val_gen_enhanced = make_gen(val_datagen, VAL_DIR, shuffle=False) test_gen_enhanced = make_gen(test_datagen, TEST_DIR, shuffle=False) NUM_CLASSES = len(train_gen_baseline.class_indices) print(f"Classes: {list(train_gen_baseline.class_indices.keys())}") print(f"Train/Val/Test: {train_gen_baseline.samples}/" f"{val_gen_baseline.samples}/{test_gen_baseline.samples}") assert train_gen_baseline.samples == 720 assert val_gen_baseline.samples == 90 assert test_gen_baseline.samples == 90 print("✓ Generators ready") # ============================================================================ # CELL 9 - 2.4.1 BASELINE ARCHITECTURE # ============================================================================ print("="*70) print("2.4.1 BASELINE ARCHITECTURE") print("="*70) tf.keras.backend.set_floatx('float32') base_b = ResNet50(weights='imagenet', include_top=False, input_shape=(IMG_SIZE, IMG_SIZE, 3)) for layer in base_b.layers[:75]: layer.trainable = False for layer in base_b.layers[75:]: layer.trainable = True frozen_b = sum(1 for l in base_b.layers if not l.trainable) trainable_b = sum(1 for l in base_b.layers if l.trainable) baseline_model = models.Sequential([ base_b, layers.GlobalAveragePooling2D(), layers.BatchNormalization(), layers.Dropout(0.5), layers.Dense(512, activation='relu', kernel_regularizer=l2(0.0001)), layers.BatchNormalization(), layers.Dropout(0.5), layers.Dense(NUM_CLASSES, activation='softmax') ], name='Baseline_ResNet50') baseline_model.compile( optimizer=tf.keras.optimizers.Adam(learning_rate=0.00005, beta_1=0.9, beta_2=0.999), loss='categorical_crossentropy', metrics=['accuracy'] ) trainable_params_baseline = sum( tf.size(w).numpy() for w in baseline_model.trainable_weights) baseline_model.summary() print(f"\nFrozen: {frozen_b} | Trainable: {trainable_b}") print(f"Trainable params: {trainable_params_baseline:,}") print("="*70) # ============================================================================ # CELL 10 - PROGRESS CALLBACK (shared by both models) # ============================================================================ class DetailedProgressCallback(tf.keras.callbacks.Callback): def on_epoch_end(self, epoch, logs=None): if (epoch + 1) % 10 == 0: print(f"\n{'='*60}") print(f"Epoch {epoch+1}:") print(f" Train: loss={logs['loss']:.4f} acc={logs['accuracy']*100:.2f}%") print(f" Val: loss={logs['val_loss']:.4f} acc={logs['val_accuracy']*100:.2f}%") print(f"{'='*60}\n") print("✓ DetailedProgressCallback ready") # ============================================================================ # CELL 11 - BASELINE CALLBACKS # ============================================================================ baseline_callbacks = [ ModelCheckpoint('baseline_best.h5', monitor='val_accuracy', save_best_only=True, mode='max', verbose=1), ReduceLROnPlateau(monitor='val_loss', factor=0.5, patience=8, min_lr=1e-7, verbose=1), DetailedProgressCallback() ] print("✓ Baseline callbacks ready (NO EarlyStopping — runs full 100 epochs)") # ============================================================================ # CELL 12 - TRAIN BASELINE MODEL # ============================================================================ print("="*70) print("TRAINING BASELINE ResNet50") print("="*70) start_baseline = time.time() history_baseline = baseline_model.fit( train_gen_baseline, validation_data=val_gen_baseline, epochs=EPOCHS, callbacks=baseline_callbacks, verbose=2 ) training_time_baseline = time.time() - start_baseline print(f"\nTraining Time: {training_time_baseline/60:.2f} min") print(f"Best Val Accuracy: {max(history_baseline.history['val_accuracy'])*100:.2f}%") print(f"Epochs: {len(history_baseline.history['loss'])}") # ============================================================================ # CELL 13 - PLOT BASELINE TRAINING CURVES # ============================================================================ fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 5)) ax1.plot(history_baseline.history['accuracy'], label='Train', lw=2, color='#2E86AB') ax1.plot(history_baseline.history['val_accuracy'], label='Validation', lw=2, color='#A23B72') ax1.axhline(0.8890, color='red', ls='--', lw=2, label='Target (88.90%)', alpha=0.7) ax1.set_title('Baseline — Accuracy', fontsize=13, fontweight='bold') ax1.set_xlabel('Epoch'); ax1.set_ylabel('Accuracy') ax1.legend(); ax1.grid(alpha=0.3) ax2.plot(history_baseline.history['loss'], label='Train', lw=2, color='#2E86AB') ax2.plot(history_baseline.history['val_loss'], label='Validation', lw=2, color='#A23B72') ax2.set_title('Baseline — Loss', fontsize=13, fontweight='bold') ax2.set_xlabel('Epoch'); ax2.set_ylabel('Loss') ax2.legend(); ax2.grid(alpha=0.3) plt.tight_layout() plt.savefig('baseline_training_history.png', dpi=300, bbox_inches='tight') plt.show() print("✓ Saved: baseline_training_history.png") # ============================================================================ # CELL 14 - ENABLE FP16 MIXED PRECISION (Enhanced only) # ============================================================================ policy = mixed_precision.Policy('mixed_float16') mixed_precision.set_global_policy(policy) print(f"✓ Mixed precision enabled: {policy.name}") print(" ~30% faster per epoch vs FP32") # ============================================================================ # CELL 15 - COMPUTE CLASS WEIGHTS (fixes Cordana bias) # ============================================================================ class_weights_array = compute_class_weight( 'balanced', classes=np.unique(train_gen_enhanced.classes), y=train_gen_enhanced.classes ) class_weight_dict = dict(enumerate(class_weights_array)) index_to_class_map = {v: k for k, v in train_gen_enhanced.class_indices.items()} print("Class Weights (fixes Cordana over-prediction):") for idx, w in class_weight_dict.items(): print(f" {index_to_class_map[idx]}: {w:.4f}") print("✓ Class weights ready") # ============================================================================ # CELL 16 - 2.4.2 ENHANCED ResNet50 ARCHITECTURE # ============================================================================ print("="*70) print("2.4.2 ENHANCED ResNet50 ARCHITECTURE") print("="*70) from tensorflow.keras import Model, Input base_e = ResNet50(weights='imagenet', include_top=False, input_shape=(IMG_SIZE, IMG_SIZE, 3)) # 80 frozen — sweet spot: more freedom than 100, stable vs 75 for layer in base_e.layers[:80]: layer.trainable = False for layer in base_e.layers[80:]: layer.trainable = True frozen_e = sum(1 for l in base_e.layers if not l.trainable) trainable_e = sum(1 for l in base_e.layers if l.trainable) inputs = Input(shape=(IMG_SIZE, IMG_SIZE, 3)) x = base_e(inputs, training=False) x = layers.BatchNormalization(name='bn_post_conv5')(x) x = layers.GlobalAveragePooling2D(name='gap')(x) x = layers.Dense(512, activation='relu', kernel_regularizer=l2(0.0005), name='dense_512')(x) x = layers.BatchNormalization(name='bn_dense1')(x) x = layers.Dropout(0.4, name='drop_04')(x) x = layers.Dense(256, activation='relu', kernel_regularizer=l2(0.0005), name='dense_256')(x) x = layers.BatchNormalization(name='bn_dense2')(x) x = layers.Dropout(0.3, name='drop_03')(x) outputs = layers.Dense(NUM_CLASSES, activation='softmax', dtype='float32', name='output')(x) enhanced_model = Model(inputs=inputs, outputs=outputs, name='Enhanced_ResNet50') trainable_params_enhanced = sum( tf.size(w).numpy() for w in enhanced_model.trainable_weights) total_params_enhanced = sum( tf.size(w).numpy() for w in enhanced_model.weights) enhanced_model.summary() print(f"\nFrozen: {frozen_e} | Trainable: {trainable_e}") print(f"Trainable params: {trainable_params_enhanced:,}") print(f"Total params: {total_params_enhanced:,}") print("="*70) # ============================================================================ # CELL 17 - COMPILE ENHANCED MODEL (AdamW) # ============================================================================ enhanced_model.compile( optimizer=tf.keras.optimizers.AdamW( learning_rate=0.0001, weight_decay=1e-4 ), loss='categorical_crossentropy', metrics=['accuracy'] ) print("✓ Enhanced compiled — AdamW | LR=0.0001 | WD=1e-4") # ============================================================================ # CELL 18 - COSINE LR SCHEDULER # ============================================================================ def cosine_decay_with_warmup(epoch, lr): initial_lr = 0.0001 min_lr = 1e-7 cosine_decay = 0.5 * (1 + np.cos(np.pi * epoch / EPOCHS)) return (initial_lr - min_lr) * cosine_decay + min_lr cosine_scheduler = LearningRateScheduler(cosine_decay_with_warmup, verbose=0) print("✓ Cosine annealing scheduler ready (0.0001 → 1e-7 over 100 epochs)") # ============================================================================ # CELL 19 - ENHANCED CALLBACKS # EarlyStopping(patience=30) = stops when converged → saves training time # restore_best_weights=True = always keeps the best checkpoint # ============================================================================ enhanced_callbacks = [ cosine_scheduler, ModelCheckpoint('enhanced_best.h5', monitor='val_accuracy', save_best_only=True, mode='max', verbose=1), tf.keras.callbacks.EarlyStopping( monitor='val_accuracy', patience=30, restore_best_weights=True, mode='max', verbose=1 ), ReduceLROnPlateau(monitor='val_loss', factor=0.5, patience=8, min_lr=1e-7, verbose=1), DetailedProgressCallback() ] print("✓ Enhanced callbacks ready") print(" EarlyStopping(patience=30) → stops at ~60-80 epochs instead of 100") print(" restore_best_weights=True → always reverts to best checkpoint") # ============================================================================ # CELL 20 - TRAIN ENHANCED MODEL # ============================================================================ print("="*70) print("TRAINING ENHANCED ResNet50") print("="*70) print("Speed vs baseline: FP16 (~30% faster/epoch) + EarlyStopping (~60-80 epochs)") print("Accuracy vs baseline: 80 frozen + class weights + AdamW + Cosine LR") print("="*70) start_enhanced = time.time() history_enhanced = enhanced_model.fit( train_gen_enhanced, validation_data=val_gen_enhanced, epochs=EPOCHS, callbacks=enhanced_callbacks, class_weight=class_weight_dict, verbose=2 ) training_time_enhanced = time.time() - start_enhanced epochs_ran = len(history_enhanced.history['loss']) best_val_acc = max(history_enhanced.history['val_accuracy']) time_red = ((training_time_baseline - training_time_enhanced) / training_time_baseline) * 100 acc_change = (best_val_acc - max(history_baseline.history['val_accuracy'])) * 100 print(f"\n{'='*70}") print("ENHANCED TRAINING COMPLETE") print(f"{'='*70}") print(f"Time: {training_time_enhanced/60:.2f} min " f"(baseline: {training_time_baseline/60:.1f} min, saved {time_red:.1f}%)") print(f"Epochs: {epochs_ran}/{EPOCHS}") print(f"Best Val: {best_val_acc*100:.2f}% " f"(baseline: {max(history_baseline.history['val_accuracy'])*100:.2f}%)") print(f"Acc change: {acc_change:+.2f} pp") if best_val_acc >= 0.8890: print("✓ SUCCESS: ≥88.90% target achieved!") elif best_val_acc > max(history_baseline.history['val_accuracy']): print("✓ IMPROVEMENT: Enhanced outperformed baseline") else: print("⚠ Below baseline — check confusion matrix after evaluation") if time_red > 0: print(f"✓ Training time reduced by {time_red:.1f}%") print("="*70) # ============================================================================ # CELL 21 - PLOT ENHANCED TRAINING CURVES # ============================================================================ fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 5)) ax1.plot(history_enhanced.history['accuracy'], label='Train', lw=2, color='#2E86AB') ax1.plot(history_enhanced.history['val_accuracy'], label='Validation', lw=2, color='#A23B72') ax1.axhline(0.8890, color='red', ls='--', lw=2, label='Target (88.90%)', alpha=0.7) ax1.axhline(max(history_baseline.history['val_accuracy']), color='orange', ls=':', lw=2, label=f"Baseline ({max(history_baseline.history['val_accuracy'])*100:.1f}%)", alpha=0.7) ax1.set_title(f'Enhanced — Accuracy (stopped epoch {epochs_ran})', fontsize=13, fontweight='bold') ax1.set_xlabel('Epoch'); ax1.set_ylabel('Accuracy') ax1.legend(); ax1.grid(alpha=0.3) ax2.plot(history_enhanced.history['loss'], label='Train', lw=2, color='#2E86AB') ax2.plot(history_enhanced.history['val_loss'], label='Validation', lw=2, color='#A23B72') ax2.set_title(f'Enhanced — Loss (stopped epoch {epochs_ran})', fontsize=13, fontweight='bold') ax2.set_xlabel('Epoch'); ax2.set_ylabel('Loss') ax2.legend(); ax2.grid(alpha=0.3) plt.tight_layout() plt.savefig('enhanced_training_history.png', dpi=300, bbox_inches='tight') plt.show() print("✓ Saved: enhanced_training_history.png") # ============================================================================ # CELL 22 - LOAD BEST MODELS FOR EVALUATION # ============================================================================ print("="*70) print("2.5 EVALUATION — loading best checkpoints") print("="*70) baseline_model = tf.keras.models.load_model('baseline_best.h5') enhanced_model = tf.keras.models.load_model('enhanced_best.h5') test_gen_eval = test_datagen.flow_from_directory( TEST_DIR, target_size=(IMG_SIZE, IMG_SIZE), batch_size=32, class_mode='categorical', shuffle=False ) print(f"Test set: {test_gen_eval.samples} images") print(f"Classes: {list(test_gen_eval.class_indices.keys())}") # ============================================================================ # CELL 23 - EVALUATION FUNCTION # ============================================================================ CLASS_DISPLAY = { 'Cordana': 'Cordana', 'Sanas': 'Healthy', 'SigatokaNegra': 'Black Sigatoka' } def evaluate_model(model, name, history, training_time, tag): print(f"\n{'='*70}") print(f"{name} — EVALUATION") print(f"{'='*70}") test_gen_eval.reset() y_probs = model.predict(test_gen_eval, verbose=0) y_pred = np.argmax(y_probs, axis=1) y_true = test_gen_eval.classes t_loss, t_acc = model.evaluate(test_gen_eval, verbose=0) print(f"\nTest Accuracy: {t_acc*100:.2f}%") print(f"Test Loss: {t_loss:.4f}") val_tmp = val_datagen.flow_from_directory( VAL_DIR, target_size=(IMG_SIZE, IMG_SIZE), batch_size=32, class_mode='categorical', shuffle=False) v_loss, v_acc = model.evaluate(val_tmp, verbose=0) print(f"Val Accuracy: {v_acc*100:.2f}%") p, r, f1, _ = precision_recall_fscore_support( y_true, y_pred, average='weighted', zero_division=0) print(f"\nPrecision (W): {p:.4f}") print(f"Recall (W): {r:.4f}") print(f"F1-Score (W): {f1:.4f}") cm = confusion_matrix(y_true, y_pred) class_names = list(test_gen_eval.class_indices.keys()) display_names= [CLASS_DISPLAY.get(c, c) for c in class_names] print("\nSpecificity per class:") for i, cls in enumerate(class_names): tn = cm.sum() - (cm[i,:].sum() + cm[:,i].sum() - cm[i,i]) fp = cm[:,i].sum() - cm[i,i] spec = tn / (tn + fp) if (tn + fp) > 0 else 0 print(f" {CLASS_DISPLAY.get(cls,cls):<20} {spec:.4f}") print("\nPer-Class Report:") print(classification_report(y_true, y_pred, target_names=display_names, digits=4)) print(f"Training Time: {training_time/60:.2f} min") print(f"Per Epoch: {training_time/len(history.history['loss']):.2f} sec") test_gen_eval.reset() t0 = time.time() _ = model.predict(test_gen_eval, verbose=0) inf_time = time.time() - OCCLUSION SENSITIVITY FUNCTIONS # ============================================================================ def occlusion_saliency_map(model, img_array, pred_idx, patch_size=48, stride=24): """~64 forward passes — ~2x faster than patch=32/stride=16.""" H, W = 224, 224 base_conf = float(model.predict(img_array, verbose=0)[0][pred_idx]) saliency = np.zeros((H, W), dtype=np.float32) counts = np.zeros((H, W), dtype=np.float32) for y in range(0, H - patch_size + 1, stride): for x in range(0, W - patch_size + 1, stride): occluded = img_array.copy() occluded[0, y:y+patch_size, x:x+patch_size, :] = 0.5 conf = float(model.predict(occluded, verbose=0)[0][pred_idx]) drop = max(0.0, base_conf - conf) saliency[y:y+patch_size, x:x+patch_size] += drop counts [y:y+patch_size, x:x+patch_size] += 1.0 counts = np.where(counts == 0, 1, counts) saliency /= counts if saliency.max() > 0: saliency /= saliency.max() return saliency def saliency_to_bbox(saliency, threshold=0.35): """Single unified box covering ALL hot regions.""" binary = (saliency >= threshold).astype(np.uint8) for t in (0.25, 0.2, 0.15, 0.1): if binary.sum() > 0: break binary = (saliency >= t).astype(np.uint8) if binary.sum() == 0: return None rows = np.any(binary, axis=1) cols = np.any(binary, axis=0) y0, y1 = np.where(rows)[0][[0, -1]] x0, x1 = np.where(cols)[0][[0, -1]] pad = 8 return (max(0, int(x0)-pad), max(0, int(y0)-pad), min(223, int(x1)+pad), min(223, int(y1)+pad)) def draw_bbox_on_image(img_pil, bbox, label, confidence, color=(220, 50, 50), lw=4): out = img_pil.copy().convert("RGB") draw = ImageDraw.Draw(out) x0, y0, x1, y1 = bbox for d in range(lw): draw.rectangle([x0-d, y0-d, x1+d, y1+d], outline=color) text = f"{label} {confidence:.1f}%" try: font = ImageFont.truetype( "/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf", 18) except: font = ImageFont.load_default() tb = draw.textbbox((0, 0), text, font=font) tw = tb[2]-tb[0]; th = tb[3]-tb[1]; pad = 4 ly = max(0, y0 - th - 2*pad - lw) draw.rectangle([x0, ly, x0+tw+2*pad, ly+th+2*pad], fill=color) draw.text((x0+pad, ly+pad), text, fill=(255, 255, 255), font=font) return out def visualize_occlusion_bbox(image_path, model, model_name="Model", patch_size=48, stride=24,
