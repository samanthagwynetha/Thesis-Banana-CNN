saliency[y:y+patch_size, x:x+patch_size] += drop counts [y:y+patch_size, x:x+patch_size] += 1.0 counts = np.where(counts == 0, 1, counts) saliency /= counts if saliency.max() > 0: saliency /= saliency.max() return saliency def saliency_to_bbox(saliency, threshold=0.4): binary = (saliency >= threshold).astype(np.uint8) if binary.sum() == 0: for t in (0.3, 0.2, 0.1): binary = (saliency >=
